function read_csv(filename)
  local rows = {}
  local f = io.open(filename, "r")
  if not f then
    tex.error("CSV file not found: " .. filename)
    return rows
  end
  for line in f:lines() do
    local row = {}
    for field in string.gmatch(line, "([^,]+)") do
      table.insert(row, field)
    end
    table.insert(rows, row)
  end
  f:close()
  return rows
end

function ResultTable(path)
  local rows = read_csv(path)
  if not rows or #rows == 0 then
    tex.print("CSV data is empty or not loaded.")
    return
  end
  local ncols = #rows[1]
  local colspec = string.rep("c", ncols)
  tex.print("\\begin{tabular}{" .. colspec .. "}")
  tex.print("\\toprule")
  for i, row in ipairs(rows) do
    local line = table.concat(row, " & ")
    tex.print(line .. " \\\\")
    if i == 1 then
      tex.print("\\midrule")
    end
  end
  tex.print("\\bottomrule")
  tex.print("\\end{tabular}")
end

local NM = {}

function NM.f(x)
  return x^4 / 81 - 1
end

function NM.df(x)
  return 4*x^3 / 81
end

function NM.newton_step(x)
  return x - NM.f(x)/NM.df(x)
end

function NM.tangent_coeff(x0)
  local m = NM.df(x0)
  local b = NM.f(x0) - m*x0
  return m, b
end

-- 図に必要なパラメータを全部まとめて返す
function NM.get_param(x0)
  local y0 = NM.f(x0)
  local x1 = NM.newton_step(x0)
  local y1 = NM.f(x1)
  local m, b = NM.tangent_coeff(x0)

  -- x は今のロジックのまま
  local margin_x = math.abs(x1 - x0) + 0.5
  local xmin = math.min(x0, x1) - margin_x
  local xmax = math.max(x0, x1) + margin_x

  local yf_min = NM.f(xmin)
  local yf_max = NM.f(xmax)

  local yt_min = m * xmin + b
  local yt_max = m * xmax + b

  local ymin = math.min(yf_min, yf_max, yt_min, yt_max)
  local ymax = math.max(yf_min, yf_max, yt_min, yt_max)

  -- 少し余白
  local margin_y = 0.1 * math.abs(ymax - ymin)
  if margin_y < 1.0 then margin_y = 1.0 end
  ymin = ymin - margin_y
  ymax = ymax + margin_y

  return {
    x0 = x0, y0 = y0,
    x1 = x1, y1 = y1, m = m, b = b,
    xmin = xmin, xmax = xmax,
    ymin = ymin, ymax = ymax,
  }
end

  tex.print(s)
end

\section{はじめに}

非線形方程式の解析的求解の限界として, 科学・工学の多くの現象は非線形方程式によって記述されるが, 三角関数や高次多項式が混在するような方程式は一般に解析的な閉じた形の解を持たない.
このような方程式に対しては, 数値的な近似解法に依らざるを得ない.\cite{narayanan2026ComparativeStudyNumerical}

根の存在保証に対する理論的要請として, 数値解法を適用する以前に, 解がどの区間に存在するかを数学的に保証する枠組みが必要とされてきた.
19世紀初頭にBolzanoが中間値の定理に基づき連続関数の零点の存在を解析的に証明した結果は, 区間内における符号変化から零点の存在を保証する原理として, 二分法のような区間縮小型アルゴリズムの理論的基盤を与えている.\cite{bernardReinAnalytischerBeweis1817}

反復法による近似解法の発展として, 根の存在が保証された後にいかに効率的に近似解へ到達するかという計算上の課題に対し, 接線近似に基づく反復解法が古くから探求されてきた.
Newtonの未刊草稿に起源を持つ手法は, 後にWallisの著作によって公刊され, 現在のNewton--Raphson法の原型として知られる反復的手続きとして確立されている.\cite{newtonAnalysiAequationesNumero1669,wallisTreatiseAlgebra1685}

収束速度の違いに関する実証的検討の必要性として, 二分法とニュートン法はいずれも非線形方程式の求解に広く用いられる基本的手法であるが, 両者は収束の速さにおいて本質的に異なる性質を持つことが理論的に知られている.
二分法は区間幅が反復ごとにほぼ半減する線形収束を示すのに対し, ニュートン法は適切な初期値と滑らかな関数のもとで誤差がほぼ2乗のオーダーで減少する二次収束を示すため, 同一の許容誤差に達するまでの反復回数に顕著な差が生じ得る.\cite{shaimbetova2025NumericalComparisonBisection}
この収束性の違いを具体的な方程式に対して定量的に比較した実測データは, 手法選択の妥当性を検証するうえで教育的にも実務的にも意義がある.

本レポートの目的は, 以上の背景を踏まえ, 非線形方程式の求解問題を1つ取り上げ, 二分法とニュートン法をそれぞれ実装し, 同一の許容誤差を満たすまでに要する反復回数を収束の速さの評価指標として両手法を定量的に比較することである.

手法の概要として, まず対象方程式のグラフを描画して解の存在区間を確認したうえで, 二分法については中間値の定理に基づく区間縮小の原理から, ニュートン法については接線近似に基づく更新式を導出し, いずれもC言語による実装を通じて各反復における近似解と誤差の推移を記録する.
これらの結果を比較指標として集計することで, 二分法とニュートン法の収束速度の違いを具体的な数値実験として明らかにする.

\section{手法}

\subsection{方程式の説明}
本レポートで対象とする方程式は以下:
\begin{equation}
  f(x)=\cos(x)-x^2
  \label{eq:fn_f}
\end{equation}
方程式(\ref{eq:fn_f})の解を求めるにあたり, まず関数$f(x)$のグラフを描くことで, 解の存在範囲を確認する.

    曲線が横軸と交わる点，すなわち $f(x)=0$ を満たす点が方程式(\ref{eq:fn_f})の解であり，

    正の側の零点 $x^*\in(0.5,1)$ に対する近似解を求めれば，対称性により負の側の零点 $-x^*$ も同時に特定できる．
    本図は，後続の二分法における初期区間 $[a_0,b_0]=[0.5,1]$ およびニュートン法における初期値 $x_0$ の選定根拠として用いられている．
  }
  \label{fig:fn_f}
\end{figure}
図\ref{fig:fn_f}に$y=f(x)$の曲線を示す. 
図\ref{fig:fn_f}より, 方程式(\ref{eq:fn_f})の解は, 区間 $0.5<x<1$ および $-1<x<-0.5$ 内にそれぞれ 1 つずつ存在することがわかる.

さらに, 関数 $f(x)=\cos(x)-x^2$ は偶関数であり, 任意の $x$ に対して $f(-x)=f(x)$ が成り立つため, グラフは $y$ 軸に関して対称である. 
したがって, 正の側に零点 $x^*>0$ が存在すれば, 負の側にはその符号を反転させた点 $-x^*$ が自動的に零点として対応する. 
本レポートではこの対称性に着目し, 数値解法の適用対象を正の側の零点 $x^*\in(0.5,1)$ 一つに限定したうえで, その近似解を求めることにする.

\subsection{非線形方程式の求解の手法}
本節では, 対象方程式(\ref{eq:fn_f})のように解析的に解けない非線形方程式に対して, 数値的に近似解を求める手法を扱う. 
ここでは, 二分法\ref{sbsbsec:bisec}とニュートン法\ref{sbsbsec:newton}について, その原理を述べたうえで漸化式またはアルゴリズムを導出する. 

\subsubsection{二分法の原理・漸化式・アルゴリズム}
\label{sbsbsec:bisec}
本節では, 連続関数 $f(x)$ に対して方程式\ref{eq:fn_f}を解く二分法を, 中間値の定理に基づく原理から説明し, 
その漸化式およびアルゴリズムを導出する. 
まず, 連続関数 $f(x)$ について, $f(a)f(b)<0$ を満たす区間 $[a,b]$ が与えられていると仮定する. 
このとき, 連続性と中間値の性質より, 端点で符号が異なることから区間内に少なくとも1つの零点 $x^*$ が存在することが保証される. 
この零点の存在に関する結果は, 19 世紀初頭に Bolzano によって解析的に証明されたことでも知られている. \cite{bernardReinAnalytischerBeweis1817}
すなわち, $a<x^*<b$ を満たす点で $f(x^*)=0$ が成り立つ. 二分法はこの「区間両端で符号が異なればその内部に零点が存在する」という原理に基づいて, 
常に零点を含む側の半区間のみを残す操作を繰り返すことで, 解の存在区間を段階的に絞り込む手法である. 
具体的には, $n$ 回目の反復で零点を含む区間 $[a_n,b_n]$ が得られているとき, 
中点を $c_n=(a_n+b_n)/2$ と定義し, 関数値 $f(c_n)$ の符号を調べる. 
もし $f(a_n)f(c_n)<0$ であれば零点は $[a_n,c_n]$ 内に存在するので, 
次の区間を $[a_{n+1},b_{n+1}]=[a_n,c_n]$ に更新する. 
そうでなければ（$f(a_n)f(c_n)\ge 0$ であれば）零点は $[c_n,b_n]$ 内に存在するので, 
$[a_{n+1},b_{n+1}]=[c_n,b_n]$ と更新する. 
関数の連続性と中間値の性質により, 各反復後の区間 $[a_{n+1},b_{n+1}]$ もなお零点を含むため, 
この操作を区間幅や関数値が所定の収束条件を満たすまで反復することで近似解を得ることができる. 
二分法では, 各反復で区間を正確に半分に縮めるため, 区間幅 $L_n=|b_n-a_n|$ は漸化式 $L_{n+1}=\tfrac{1}{2}L_n$ を満たす. 
初期区間幅を $L_0=|b_0-a_0|$ とおくと, これを繰り返し適用することで $L_n=\dfrac{|b_0-a_0|}{2^n}$ と表される. 
したがって, $n$ 回目の反復後の区間幅は $|b_n-a_n|=\dfrac{|b_0-a_0|}{2^n}$ となり, 目標とする精度に必要な反復回数をこの式から見積もることができる. 

\subsubsection{ニュートン法の原理・漸化式・アルゴリズム}
\label{sbsbsec:newton}
本節では, 連続微分可能な関数 $f(x)$ に対して方程式~\eqref{eq:fn_f} を解くニュートン法を, 接線近似に基づく原理から説明し, 
その漸化式およびアルゴリズムを導出する. 
ニュートン法の起源は, Isaac Newton による未刊草稿\emph{De Analysi per Aequationes Numero Terminorum Infinitas}（1669）にさかのぼり, \cite{newtonAnalysiAequationesNumero1669}
その手法が初めて印刷物として広く公刊されたのはWallis の代数学書 \emph{A Treatise of Algebra, both Historical and Practical}（1685）とされる. \cite{wallisTreatiseAlgebra1685}
同書では方程式の近似解を求める反復的手続きが述べられており, 後に Newton--Raphson 法として確立された方法の原型が見られる. 

いま, 求めたい解 $x^*$ が $f(x^*)=0$ を満たすとする. ニュートン法は, 
現時点の近似解 $x_n$ の近傍で $f(x)$ を 1 次のテイラー展開によって線形近似し, 
この直線近似の零点を次の近似値として更新するという原理に基づく. 
すなわち, $x_n$ の近傍で(\ref{eq:newt_approx})式と近似し, 右辺を 0 とおいて解くと, (\ref{eq:newt_recur_rearng})式より, (\ref{eq:newt_recur})式という漸化式が得られる. 
\begin{equation}
  \label{eq:newt_approx}
  f(x) \approx f(x_n) + f'(x_n)(x-x_n)
\end{equation}
\begin{equation}
  \label{eq:newt_recur_rearng}
  f(x_n) + f'(x_n)(x_{n+1}-x_n) = 0
\end{equation}
\begin{equation}
  \label{eq:newt_recur}
  x_{n+1} = x_n - \frac{f(x_n)}{f'(x_n)}
\end{equation}
この更新式は「現在の点 $x_n$ におけるグラフの接線を $x$-軸と交わるまで延長し, 
その交点を次の近似解 $x_{n+1}$ とする」という幾何学的解釈を持ち, 
Wallis の著作でもこのような接線に基づく逐次近似の考え方が現れている.\cite{wallisTreatiseAlgebra1685}

一般に, 初期値 $x_0$ の選び方と関数 $f$ の性質に適切な条件（例えば, 
$x^*$ の近傍で $f$ が十分滑らかであり, $f'(x^*)\neq 0$ を満たすこと）を課すとき, 
ニュートン法の反復列 $\{x_n\}$ は真の解 $x^*$ に向かって収束し, 
さらに誤差はおおよそ 2 乗で減少する, すなわち 2 次収束することが知られている. 
具体的なアルゴリズムとしては, 初期値 $x_0$ を与えたのち, 
各反復 $n=0,1,2,\dots$ で\ref{eq:newt_step}式
  \begin{equation}
    \label{eq:newt_step}
    x_{n+1}=x_n-\frac{f(x_n)}{f'(x_n)}
  \end{equation}
を計算し, $|x_{n+1}-x_n|$ や $|f(x_{n+1})|$ が所定の収束判定条件を満たした時点で反復を終了し, $x_{n+1}$ を近似解とみなす. 
このようにニュートン法は, 接線近似という単純な幾何学的原理に基づきつつ, 
適切な条件のもとでは誤差が2乗のオーダーで減少する2次収束を達成する反復解法である.

\subsection{プログラムの仕様}
\subsubsection{二分法の仕様}
入力： 関数 $f$，初期区間 $[a,b]$（$f(a)f(b)<0$），収束判定の許容誤差 $\varepsilon$．
出力： 方程式 $f(x)=0$ の近似解 $\hat{x}$ と，そこに至るまでの反復回数 $n$．
機能： 中点 $c=(a+b)/2$ を計算し，誤差指標として半区間幅 $|b-a|/2$ を用いる．
  $|b-a|/2>\varepsilon$ の間，零点を含む側の半区間のみを選択して区間を半分に狭める反復を行う．
  収束条件を満たした時点の中点 $c$ を近似解 $\hat{x}$ とし，各反復における反復回数と半区間幅，
  近似解を CSV 形式で出力する．

\subsubsection{ニュートン法の仕様}
入力： 関数 $f$ とその導関数 $f'$，初期値 $x_0$，収束判定の許容誤差 $\varepsilon$，最大反復回数 $N$．
出力： 方程式 $f(x)=0$ の近似解 $\hat{x}$ と，そこに至るまでの反復回数 $n$．
機能： 
  初期値 $x_0$ から始めて(\ref{eq:newt_step})式に従う更新反復を行い，
  更新量の絶対値 $|f(x_n)/f'(x_n)|<\varepsilon$ となるか $n\ge N$ に達するまで近似解を更新する．
  収束条件を満たした時点の値を近似解 $\hat{x}$ とし，各反復における反復回数，
  更新量の大きさ，近似解を CSV 形式で出力する．

\subsubsection{実行環境}
本報告書で用いた二分法およびニュートン法の C プログラムは，
macOS バージョン 26.5.1（ビルド 25F80）上で実行した．
コンパイルには標準的な C コンパイラ \texttt{gcc} を用い，
数値計算には C 言語の標準ライブラリ \texttt{math.h} の関数
\texttt{cos}, \texttt{sin}, \texttt{pow} などのみを利用した．
外部ライブラリや特別な数値計算パッケージには依存していないため，
同等の C コンパイラと標準ライブラリが利用できる環境であれば，同様の結果が再現可能である．

\subsection{評価指標}
本報告書では, 非線形方程式の近似解を与える両手法の収束の速さを, 許容誤差を満たすまでに要した反復回数によって評価する. 
具体的には, 正の側の零点 $x^*$ に対する近似解が許容誤差 $\varepsilon = 10^{-5}$ を満たすまで反復を続けたときの反復回数 $n$ を収束の速さの評価指標として一意に定める.
二分法とニュートン法を同一条件下で比較するため, 両手法とも同じ方程式 $f(x)=\cos(x)-x^2$ を対象とし,
同じ零点 $x^*$ に収束するように初期区間および初期値を設定したうえで, この評価指標 $n$ を用いて結果を集計する.

\section{結果}
本節では，二分法およびニュートン法を方程式 $f(x)=\cos(x)-x^2$ に適用して得られた近似解と，
許容誤差 $\varepsilon = 10^{-5}$ を満たすまでの反復過程を示す．

まず二分法について，反復回数 $n=1$ から $n=16$ までの各段階で得られた
近似解および誤差の値を表\ref{tab:bi_result}に示す．
表\ref{tab:bi_result}より，近似解は反復ごとに $x^*\approx 0.824142$ へ
単調に近づいており，誤差指標として用いた半区間幅 $|b-a|/2$ は
初期の $2.5\times 10^{-1}$ から反復ごとに約半分ずつ減少し，
16 回目には約 $7.6\times 10^{-6}$ となり，許容誤差 $\varepsilon$ を
十分に下回っていることが確認できる．

\section{おわりに}

本レポートでは, 超越方程式 $f(x)=\cos(x)-x^2$ に対して二分法とニュートン法を同一条件下で適用し, 収束速度を反復回数により比較することで, 既存の理論的知見が具体的な数値実験としてどの程度再現されるかを検討した. 
結果節で示したように, 許容誤差 $\varepsilon=10^{-5}$ を満たすまでに要した反復回数は二分法で $n=16$, ニュートン法で $n=5$ となり, 線形収束を示す二分法と二次収束を示すニュートン法の性能差が, 1 つの方程式に対する実験的データとして明瞭に確認された.\cite{pandey2024ComparisonNumericalAccuracy,embree2020Lecture30}

この結果は, 一般的な数値解析の教科書や講義ノートで述べられている「二分法は常に収束するが収束速度は線形で遅く, ニュートン法は初期値が十分良ければ二次収束により高速に収束する」という整理と整合的である.\cite{embree2020Lecture30,gavva2020NUMERICALANALYSIS}
一方で, ニュートン法の高速性は, 初期値が真の解の近傍にあり, かつ対象の零点が単純根であるという条件のもとでのみ保証されることが知られており, 初期値が不適切な場合や多重根に対しては収束が遅くなったり失敗したりする可能性がある.\cite{embree2020Lecture30}
本レポートでは初期値をグラフと関数の対称性に基づいて慎重に選んだため, ニュートン法が理論どおり二次収束を示す好条件のケースに限定されている点は, 今回の検討の前提条件として明示しておく必要がある. 

既存研究との比較という観点からは, Pandey と Singh による, 4 種類の根探しアルゴリズムの数値精度を C++ 実装により比較した研究がある.\cite{pandey2024ComparisonNumericalAccuracy}
同研究では, 二分法, 偽位置法, ニュートン法, セカント法を用いて 1 から 30 までの 4 乗根を計算し, 平均パーセント誤差を指標として手法間の精度差を評価しているが, ニュートン法とセカント法が二分法より小さな誤差で高精度な近似解を与えることが示されており, 高次の収束次数を持つ手法の優位性という点で本レポートの結果と整合的である.\cite{pandey2024ComparisonNumericalAccuracy}
また Embree によるニュートン法の収束解析の講義ノートでは, 二分法の線形収束率が $1/2$, ニュートン法の二次収束における誤差の漸化式と漸近誤差定数が明示的に導かれており, 本レポートの反復回数の差はこれらの理論式の具体例とみなせる.\cite{embree2020Lecture30}

今回の数値実験の問題点としては, 対象方程式を 1 つの実関数に限定し, かつニュートン法の初期値選択をグラフから視覚的に行ったため, 初期値が不適切な場合の収束失敗や, 多重根に対する収束次数の変化といった, ニュートン法の弱点を十分には評価していない点が挙げられる.\cite{embree2020Lecture30} 
また, 二分法とニュートン法のみを比較対象とし, 偽位置法やセカント法, あるいは収束性の改善を意図した修正ニュートン法など, 他の根探しアルゴリズムとの系統的な比較を行っていないため, 「二分法 vs ニュートン法」という二者択一の枠組みの中でのみ議論が閉じている点も今後の改善余地として残る.\cite{pandey2024ComparisonNumericalAccuracy}

今後の課題としては, 本レポートで扱った一次元の実関数に対する根探しを, 多変数関数や連立非線形方程式に対する解法へ一般化し, 多変数ニュートン法やその修正版の収束性とロバスト性を実験的に比較することが考えられる.\cite{gavva2020NUMERICALANALYSIS} 
さらに, Pandey らの研究で扱われている偽位置法やセカント法などの他アルゴリズムを含めて, 同一の評価指標のもとで収束速度と数値精度, 初期値依存性, 実装の容易さなどを総合的に比較し, 理論的な収束次数だけでなく実務的な観点からの「使いやすさ」も含めて根探しアルゴリズムの特性を整理することが, 次の段階で取り組むべき課題である.\cite{pandey2024ComparisonNumericalAccuracy}
